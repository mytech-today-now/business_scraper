🚀 Deployment Commands

Quick Start

# Start production stack
docker compose -f docker-compose.simple-prod.yml --env-file .env.docker up -d

# Check status
docker ps

# View logs
docker logs business-scraper-app

# Health check
curl http://localhost:3000/api/health

# Run monitoring
./monitoring/health-check.sh

Management Commands

# Stop services
docker compose -f docker-compose.simple-prod.yml down

# Update application
npm run build
docker compose -f docker-compose.simple-prod.yml up -d --build

# Backup database
docker exec business-scraper-db pg_dump -U postgres business_scraper > backup.sql
